# MQTT PUBLISHER 
This package provides an easy-to-use and customizable MQTT client for communication with MQTT servers.

# .ENV EXAMPLE
MQTT_PORT=1883
MQTT_HOST=localhost
MQTT_USER=user
MQTT_PASSWORD=password
MQTT_CLIENT_ID=123456789
MQTT_CLEAN=true
MQTT_CLIENT_TIMEOUT=4000
MQTT_RECONNECT_PERIOD=1000
MQTT_KEEP_ALIVE=60
MQTT_ENV=dev
